Quick Commands

🛠️ Useful Commands

# Check deployments status
kubectl get deployments -n bangla-market-staging
# View helm releases
helm list -n bangla-market-staging
# Check pod logs
kubectl logs -f deployment/api-gateway -n bangla-market-staging
# Manual deployment update
helm upgrade bangla-market-staging ./helm/bangla-market -f values.staging.yaml
# Scale deployment
kubectl scale deployment api-gateway --replicas=3 -n bangla-market-staging
